88QString SystemManager::fetchWifiStatus(QString &wifiName) const {
89 QProcess process;
90 process.start("nmcli", {"-t", "-f", "DEVICE,STATE,CONNECTION", "dev"});
91 process.waitForFinished();
92
93 QString output = process.readAllStandardOutput().trimmed();
94 QStringList lines = output.split('\n');
95
96 for (const QString &line : lines) {
97 if (line.startsWith("wlan")) { // Assuming WiFi interface starts with 'wlan'
98 QStringList parts = line.split(':');
99 if (parts.size() >= 3) {
100 QString state = parts[1];
101 wifiName = parts[2]; // Extract connection name
102
103 if (state == "connected") {
104 return "Connected";
105 } else {
106 wifiName.clear();
107 return "Disconnected";
108 }
109 }
110 }
111 }
112 wifiName.clear();
113 return "No interface detected";
114}

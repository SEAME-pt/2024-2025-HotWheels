37std::vector<LanePostProcessor::LaneInfo> LanePostProcessor::extractLaneInfo(const cv::Mat& mask, bool filterSize, bool filterLength) {
38 std::vector<std::vector<cv::Point>> contours;
39 cv::findContours(mask, contours, cv::RETR_EXTERNAL, cv::CHAIN_APPROX_SIMPLE);
40
41 std::vector<LaneInfo> lanes;
42 int height = mask.rows, width = mask.cols;
43
44 for (const auto& cnt : contours) {
45 double area = cv::contourArea(cnt);
46 double length = cv::arcLength(cnt, false);
47
48 if ((!filterSize || area > minComponentSize) &&
49 (!filterLength || length > minComponentLength) &&
50 cnt.size() >= 2) {
51

54 float vx = line[0], vy = line[1], x0 = line[2], y0 = line[3];

56
57 cv::Moments M = cv::moments(cnt);
58 if (M.m00 == 0) continue;
59 cv::Point centroid(M.m10 / M.m00, M.m01 / M.m00);
60
61 std::vector<cv::Point> intersections;
62 if (vy != 0) {
63 float t = -y0 / vy;
64 float x = x0 + vx * t;
65 if (x >= 0 && x < width) intersections.emplace_back(x, 0);
66 t = (height - 1 - y0) / vy;
67 x = x0 + vx * t;
68 if (x >= 0 && x < width) intersections.emplace_back(x, height - 1);
69 }
70 if (vx != 0) {
71 float t = -x0 / vx;
72 float y = y0 + vy * t;
73 if (y >= 0 && y < height) intersections.emplace_back(0, y);
74 t = (width - 1 - x0) / vx;
75 y = y0 + vy * t;
76 if (y >= 0 && y < height) intersections.emplace_back(width - 1, y);
77 }
78
79 if (intersections.size() >= 2) {
80 lanes.push_back({ static_cast<int>(lanes.size() + 1), angle, centroid, {intersections[0], intersections[1]}, cnt });

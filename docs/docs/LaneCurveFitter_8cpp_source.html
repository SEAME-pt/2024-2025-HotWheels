21float interpolateXatY(const std::vector<cv::Point2f>& points, float y_query) {
22 if (points.empty()) return 0.0f;
23
24 for (size_t i = 1; i < points.size(); ++i) {
25 float y1 = points[i - 1].y;
26 float y2 = points[i].y;
27
28 if ((y1 <= y_query && y_query <= y2) || (y2 <= y_query && y_query <= y1)) {
29 float t = (y_query - y1) / (y2 - y1 + 1e-6f);
30 float x1 = points[i - 1].x;
31 float x2 = points[i].x;
32 return x1 + t * (x2 - x1);
33 }
34 }
35
36 // Extrapolate if y_query is outside range
37 return points.back().x;
38}

41std::vector<int> LaneCurveFitter::dbscanCluster(const std::vector<cv::Point>& points, std::vector<int>& uniqueLabels) {
42 const int n = points.size();
43 std::vector<int> labels(n, -1);
44 int clusterId = 0;
45
46 for (int i = 0; i < n; ++i) {
47 if (labels[i] != -1) continue;
48
49 std::vector<int> neighbors;
50 for (int j = 0; j < n; ++j) {
51 if (cv::norm(points[i] - points[j]) <= dbscanEps)
52 neighbors.push_back(j);
53 }
54
55 if (neighbors.size() < dbscanMinSamples)
56 continue;
57
58 labels[i] = clusterId;
59 std::set<int> seeds(neighbors.begin(), neighbors.end());
60 seeds.erase(i);
61
62 while (!seeds.empty()) {
63 int current = *seeds.begin();
64 seeds.erase(seeds.begin());
65
66 if (labels[current] == -1) {
67 labels[current] = clusterId;
68
69 std::vector<int> currentNeighbors;
70 for (int j = 0; j < n; ++j) {
71 if (cv::norm(points[current] - points[j]) <= dbscanEps)
72 currentNeighbors.push_back(j);
73 }
74
75 if (currentNeighbors.size() >= dbscanMinSamples) {
76 seeds.insert(currentNeighbors.begin(), currentNeighbors.end());
77 }
78 }
79 }
80
81 ++clusterId;
82 }
83
84 uniqueLabels.clear();
85 for (int l : labels)
86 if (l != -1)
87 uniqueLabels.push_back(l);
88 std::sort(uniqueLabels.begin(), uniqueLabels.end());
89 uniqueLabels.erase(std::unique(uniqueLabels.begin(), uniqueLabels.end()), uniqueLabels.end());
90
91 return labels;

97

99 int yLow = imgSize.height - (i + 1) * h;

125bool LaneCurveFitter::isStraightLine(const std::vector<float>& y, const std::vector<float>& x, float threshold) {
126 if (x.size() < 4) return false;
127
128 float mean_x = std::accumulate(x.begin(), x.end(), 0.0f) / x.size();
129 float mean_y = std::accumulate(y.begin(), y.end(), 0.0f) / y.size();
130
131 float num = 0.0f, den_x = 0.0f, den_y = 0.0f;
132 for (size_t i = 0; i < x.size(); ++i) {
133 num += (x[i] - mean_x) * (y[i] - mean_y);
134 den_x += (x[i] - mean_x) * (x[i] - mean_x);
135 den_y += (y[i] - mean_y) * (y[i] - mean_y);
136 }
137
138 float corr = num / std::sqrt(den_x * den_y + 1e-6f);
139 return std::abs(corr) > threshold;
140}

142bool LaneCurveFitter::hasSignFlip(const std::vector<float>& xVals) {
143 std::vector<float> dx2(xVals.size());
144 for (size_t i = 1; i + 1 < xVals.size(); ++i)
145 dx2[i] = xVals[i + 1] + xVals[i - 1] - 2 * xVals[i];
146
147 for (size_t i = 1; i < dx2.size(); ++i)
148 if ((dx2[i] > 0) != (dx2[i - 1] > 0))
149 return true;
150 return false;

153std::vector<float> LaneCurveFitter::fitCurve(const std::vector<float>& y, const std::vector<float>& x, const std::vector<float>& yEval) {
154 if (y.size() < 3 || x.size() < 3) {
155 // Fallback: return a straight horizontal line
156 std::vector<float> fallback(yEval.size(), x.empty() ? 0.0f : x[0]);
157 return fallback;
158 }
159
160 cv::Mat A(y.size(), 3, CV_32F);
161 cv::Mat X(x);
162
163 for (size_t i = 0; i < y.size(); ++i) {
164 A.at<float>(i, 0) = y[i] * y[i];
165 A.at<float>(i, 1) = y[i];
166 A.at<float>(i, 2) = 1.0f;
167 }
168
169 cv::Mat coeffs;
170 if (!cv::solve(A, X, coeffs, cv::DECOMP_SVD)) {
171 // Fallback in case of failure
172 std::vector<float> fallback(yEval.size(), x[0]);
173 return fallback;
174 }
175
176 std::vector<float> result;
177 for (float yv : yEval) {
178 result.push_back(coeffs.at<float>(0) * yv * yv + coeffs.at<float>(1) * yv + coeffs.at<float>(2));
179 }
180 return result;
181}

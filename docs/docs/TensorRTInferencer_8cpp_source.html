189void TensorRTInferencer::runInference(const cv::cuda::GpuMat& gpuInput) {
190 if (gpuInput.rows != inputSize.height || gpuInput.cols != inputSize.width) { // Verify input dimensions
191 throw std::runtime_error("Input image dimensions do not match expected dimensions");
192 }
193
194 cudaError_t err = cudaMemcpy2DAsync(
195 deviceInput, // Destination: TensorRT input buffer
196 inputSize.width * sizeof(float), // Destination row stride
197 gpuInput.ptr<float>(), // Source pointer: GpuMat data
198 gpuInput.step, // Source stride
199 inputSize.width * sizeof(float), // Width to copy in bytes
200 inputSize.height, // Height to copy (rows)
201 cudaMemcpyDeviceToDevice, // Type of copy: GPU to GPU
202 stream // Use CUDA stream
203 );
204
205 if (err != cudaSuccess) { // Check if memory copy failed
206 throw std::runtime_error("cudaMemcpy2DAsync failed: " + std::string(cudaGetErrorString(err)));
207 }
208
209 if (!context->enqueueV2(bindings.data(), stream, nullptr)) { // Enqueue inference on the GPU

96std::vector<float> YOLOv5TRT::infer(const cv::Mat& image) {
97 // Usar GPU para processamento
98 gpu_image.upload(image);
99 cv::cuda::resize(gpu_image, gpu_resized, cv::Size(640, 640));
100 gpu_resized.convertTo(gpu_float, CV_32FC3, 1.0/255.0);
101
102 // Download otimizado
103 gpu_float.download(blob);
104 cv::split(blob, channels);
105
106 // Cópia otimizada dos canais (HWC -> CHW)
107 for (int c = 0; c < 3; c++) {
108 memcpy(hostDataBuffer + c*640*640,
109 channels[c].ptr<float>(),
110 640*640*sizeof(float));
111 }
112
113 // Copiar dados para GPU
114 cudaMemcpyAsync(inputDevice, hostDataBuffer, 3*640*640*sizeof(float),
115 cudaMemcpyHostToDevice, stream);
116
117 // Executar inferência
118 context->enqueueV2(bindings.data(), stream, nullptr);
119
120 // Copiar resultados para o host
121 cudaMemcpyAsync(outputHost, outputDevice, outputSize,
122 cudaMemcpyDeviceToHost, stream);
123 cudaStreamSynchronize(stream);
124
125 return std::vector<float>(outputHost, outputHost + outputSize / sizeof(float));
126}

136std::vector<Detection> YOLOv5TRT::postprocess(const std::vector<float>& output, int num_classes, float conf_thresh, float nms_thresh) {
137 std::vector<Detection> dets;
138 int num_preds = output.size() / (5 + num_classes);
139
140 for (int i = 0; i < num_preds; ++i) {
141 const float* pred = &output[i * (5 + num_classes)];
142 float obj = pred[4];
143 if (obj < conf_thresh) continue;
144
145 // Encontrar a classe com maior probabilidade
146 float max_cls = pred[5];
147 int class_id = 0;
148 for (int c = 1; c < num_classes; ++c) {
149 if (pred[5 + c] > max_cls) {
150 max_cls = pred[5 + c];
151 class_id = c;
152 }
153 }
154
155 float score = obj * max_cls;
156 if (score < conf_thresh) continue;
157
158 dets.push_back({pred[0], pred[1], pred[2], pred[3], score, class_id});
159 }
160
161 // NMS
162 std::vector<Detection> result;
163 std::sort(dets.begin(), dets.end(), [](const Detection& a, const Detection& b) {
164 return a.conf > b.conf;
165 });
166
167 std::vector<bool> removed(dets.size(), false);
168 for (size_t i = 0; i < dets.size(); ++i) {
169 if (removed[i]) continue;
170 result.push_back(dets[i]);
171
172 for (size_t j = i + 1; j < dets.size(); ++j) {
173 if (removed[j]) continue;
174
175 // Calcular IoU
176 float xx1 = std::max(dets[i].x - dets[i].w/2, dets[j].x - dets[j].w/2);
177 float yy1 = std::max(dets[i].y - dets[i].h/2, dets[j].y - dets[j].h/2);
178 float xx2 = std::min(dets[i].x + dets[i].w/2, dets[j].x + dets[j].w/2);
179 float yy2 = std::min(dets[i].y + dets[i].h/2, dets[j].y + dets[j].h/2);
180
181 float w = std::max(0.0f, xx2 - xx1);
182 float h = std::max(0.0f, yy2 - yy1);
183 float inter = w * h;
184 float area1 = dets[i].w * dets[i].h;
185 float area2 = dets[j].w * dets[j].h;
186 float ovr = inter / (area1 + area2 - inter);
187
188 if (ovr > nms_thresh) removed[j] = true;
189 }
190 }
191 return result;
192}

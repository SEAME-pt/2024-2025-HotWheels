145TEST_F(MCP2515ControllerTest, ProcessReadingCallsHandlers) {

147 EXPECT_CALL(mockSPI, closeDevice()).Times(1);
148 EXPECT_CALL(mockSPI, readByte(_))
149 .WillOnce(Return(0x01)) // Indicate data available
150 .WillRepeatedly(Return(0x00)); // No more data
151 EXPECT_CALL(mockSPI, spiTransfer(_, _, _))
152 .WillRepeatedly([](const uint8_t *tx, uint8_t *rx, size_t length) {
153 if (length == 3 && tx[0] == 0x03) { // Read command
154 rx[1] = 0x12; // Frame ID part 1
155 rx[2] = 0x34; // Frame ID part 2
156 }
157 });
158 EXPECT_CALL(mockSPI, writeByte(_, _)).Times(::testing::AtLeast(1));
159
160 MCP2515Controller controller("/dev/spidev0.0", mockSPI);
161
162 std::thread readerThread([&controller]() { controller.processReading(); });
163
164 std::this_thread::sleep_for(std::chrono::milliseconds(50));
165 controller.stopReading();

168

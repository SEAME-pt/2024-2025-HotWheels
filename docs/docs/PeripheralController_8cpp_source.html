60PeripheralController::PeripheralController(int servo_addr, int motor_addr)
61 : servo_addr_(servo_addr), motor_addr_(motor_addr) {
62 // Initialize I2C buses
63 servo_bus_fd_ = open("/dev/i2c-1", O_RDWR);
64 motor_bus_fd_ = open("/dev/i2c-1", O_RDWR);
65
66 if (servo_bus_fd_ < 0 || motor_bus_fd_ < 0) {
67 throw std::runtime_error("Failed to open I2C device");
68 return;
69 }
70
71 // Set device addresses
72 if (ioctl(servo_bus_fd_, I2C_SLAVE, servo_addr_) < 0 ||
73 ioctl(motor_bus_fd_, I2C_SLAVE, motor_addr_) < 0) {
74 throw std::runtime_error("Failed to set I2C address");
75 return;
76 }
77}

95int PeripheralController::i2c_smbus_write_byte_data(int file, uint8_t command,

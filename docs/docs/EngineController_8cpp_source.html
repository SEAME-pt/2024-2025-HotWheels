110void EngineController::set_speed(int speed) {
111
112 speed = clamp(speed, -100, 100);
113 int pwm_value = static_cast<int>(std::abs(speed) / 100.0 * 4096);
114
115 if (speed >
116 0) { // Forward (but actually backward because joysticks are reversed)
117 pcontrol->set_motor_pwm(0, pwm_value);
118 pcontrol->set_motor_pwm(1, 0);
119 pcontrol->set_motor_pwm(2, pwm_value);
120 pcontrol->set_motor_pwm(5, pwm_value);
121 pcontrol->set_motor_pwm(6, 0);
122 pcontrol->set_motor_pwm(7, pwm_value);
123 setDirection(CarDirection::Reverse);
124 } else if (speed < 0) { // Backwards
125 pcontrol->set_motor_pwm(0, pwm_value);
126 pcontrol->set_motor_pwm(1, pwm_value);
127 pcontrol->set_motor_pwm(2, 0);
128 pcontrol->set_motor_pwm(5, 0);
129 pcontrol->set_motor_pwm(6, pwm_value);
130 pcontrol->set_motor_pwm(7, pwm_value);
131 setDirection(CarDirection::Drive);
132 } else { // Stop
133 for (int channel = 0; channel < 9; ++channel)
134 pcontrol->set_motor_pwm(channel, 0);
135 setDirection(CarDirection::Stop);
136 }
137 m_current_speed = speed;
138}

145void EngineController::set_steering(int angle) {
146 angle = clamp(angle, -MAX_ANGLE, MAX_ANGLE);
147 int pwm = 0;
148 if (angle < 0) {
149 pwm = SERVO_CENTER_PWM +
150 static_cast<int>((angle / static_cast<float>(MAX_ANGLE)) *
151 (SERVO_CENTER_PWM - SERVO_LEFT_PWM));
152 } else if (angle > 0) {
153 pwm = SERVO_CENTER_PWM +
154 static_cast<int>((angle / static_cast<float>(MAX_ANGLE)) *
155 (SERVO_RIGHT_PWM - SERVO_CENTER_PWM));
156 } else {
157 pwm = SERVO_CENTER_PWM;
158 }
159
160 pcontrol->set_servo_pwm(STEERING_CHANNEL, 0, pwm);
161 m_current_angle = angle;
162 emit this->steeringUpdated(angle);
163}

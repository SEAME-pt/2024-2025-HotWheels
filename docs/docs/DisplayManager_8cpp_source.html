144QString getWifiSSID() {
145 // Linux: read active SSID using shell command
146 QProcess proc;
147 proc.start("iwgetid -r"); // Gets current SSID
148 proc.waitForFinished();
149 QString output = proc.readAllStandardOutput().trimmed();
150 return output.isEmpty() ? "Not connected" : output;
151}

153QString getLocalIPAddress() {
154 const QList<QHostAddress> &addresses = QNetworkInterface::allAddresses();
155 for (const QHostAddress &address : addresses) {
156 if (address.protocol() == QAbstractSocket::IPv4Protocol && !address.isLoopback())
157 return address.toString();
158 }
159 return "No IP";
160}

225void DisplayManager::updateDrivingMode(DrivingMode newMode) {
226 QString modeText;
227 switch (newMode) {
228 case DrivingMode::Manual:
229 modeText = "Manual";
230 m_ui->laneKeepingAssistLabel->show();
231 m_ui->laneDepartureWarningLabel->show();
232 // Stop blinking if active
233 if (m_blinkTimer) {
234 m_blinkTimer->stop();
235 m_blinkTimer->deleteLater();
236 m_blinkTimer = nullptr;
237 }
238 break;
239 case DrivingMode::Automatic:
240 modeText = "Automatic";
241 m_ui->laneKeepingAssistLabel->hide();
242 if (!m_blinkTimer) {
243 m_blinkTimer = new QTimer(this);
244 connect(m_blinkTimer, &QTimer::timeout, this, [=]() {
245 bool currentlyVisible = m_ui->laneDepartureWarningLabel->isVisible();
246 m_ui->laneDepartureWarningLabel->setVisible(!currentlyVisible);
247 });
248 m_blinkTimer->start(150); // Blink every 150ms
249 }
250 break;
251 }
252 m_ui->drivingModeLabel->setText(modeText);
253}

304void DisplayManager::displayInferenceImage(const QImage &image) {
305 if (!m_ui->inferenceLabel)
306 return;
307
308 QPixmap original = QPixmap::fromImage(image);
309 QPixmap rounded(original.size());
310 rounded.fill(Qt::transparent);
311
312 QPainter painter(&rounded);
313 painter.setRenderHint(QPainter::Antialiasing, true);
314
315 QPainterPath path;
316 path.addRoundedRect(original.rect(), 34, 34);
317 painter.setClipPath(path);
318 painter.drawPixmap(0, 0, original);
319

116void EngineController::set_speed(int speed) {
117
118 speed = clamp(speed, -100, 100);
119 int pwm_value = static_cast<int>(std::abs(speed) / 100.0 * 4096);
120
121 if (speed >
122 0) { // Forward (but actually backward because joysticks are reversed)
123 pcontrol->set_motor_pwm(0, pwm_value);
124 pcontrol->set_motor_pwm(1, 0);
125 pcontrol->set_motor_pwm(2, pwm_value);
126 pcontrol->set_motor_pwm(5, pwm_value);
127 pcontrol->set_motor_pwm(6, 0);
128 pcontrol->set_motor_pwm(7, pwm_value);
129 setDirection(CarDirection::Reverse);
130 } else if (speed < 0) { // Backwards
131 pcontrol->set_motor_pwm(0, pwm_value);
132 pcontrol->set_motor_pwm(1, pwm_value);
133 pcontrol->set_motor_pwm(2, 0);
134 pcontrol->set_motor_pwm(5, 0);
135 pcontrol->set_motor_pwm(6, pwm_value);
136 pcontrol->set_motor_pwm(7, pwm_value);
137 setDirection(CarDirection::Drive);
138 } else { // Stop
139 for (int channel = 0; channel < 9; ++channel)
140 pcontrol->set_motor_pwm(channel, 0);
141 setDirection(CarDirection::Stop);
142 }
143 m_current_speed = speed;
144}

153void EngineController::set_steering(int angle) {
154 angle = clamp(angle, -MAX_ANGLE, MAX_ANGLE);
155 int pwm = 0;
156 if (angle < 0) {
157 pwm = SERVO_CENTER_PWM +
158 static_cast<int>((angle / static_cast<float>(MAX_ANGLE)) *
159 (SERVO_CENTER_PWM - SERVO_LEFT_PWM));
160 } else if (angle > 0) {
161 pwm = SERVO_CENTER_PWM +
162 static_cast<int>((angle / static_cast<float>(MAX_ANGLE)) *
163 (SERVO_RIGHT_PWM - SERVO_CENTER_PWM));
164 } else {
165 pwm = SERVO_CENTER_PWM;
166 }
167
168 pcontrol->set_servo_pwm(STEERING_CHANNEL, 0, pwm);
169 m_current_angle = angle;
170 emit this->steeringUpdated(angle);
171}

65PeripheralController::PeripheralController(int servo_addr, int motor_addr)
66 : servo_addr_(servo_addr), motor_addr_(motor_addr) {
67 // Initialize I2C buses
68 servo_bus_fd_ = open("/dev/i2c-1", O_RDWR);
69 motor_bus_fd_ = open("/dev/i2c-1", O_RDWR);
70
71 if (servo_bus_fd_ < 0 || motor_bus_fd_ < 0) {
72 throw std::runtime_error("Failed to open I2C device");
73 return;
74 }
75
76 // Set device addresses
77 if (ioctl(servo_bus_fd_, I2C_SLAVE, servo_addr_) < 0 ||
78 ioctl(motor_bus_fd_, I2C_SLAVE, motor_addr_) < 0) {
79 throw std::runtime_error("Failed to set I2C address");
80 return;
81 }
82}

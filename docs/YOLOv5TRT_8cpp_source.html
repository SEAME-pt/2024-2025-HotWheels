127 std::vector<Detection> YOLOv5TRT::postprocess(const std::vector<float>& output, int num_classes, float conf_thresh, float nms_thresh) {
128  std::vector<Detection> dets;
129  int num_preds = output.size() / (5 + num_classes);
130 
131  for (int i = 0; i < num_preds; ++i) {
132  const float* pred = &output[i * (5 + num_classes)];
133  float obj = pred[4];
134  if (obj < conf_thresh) continue;
135 
136  // Encontrar a classe com maior probabilidade
137  float max_cls = pred[5];
138  int class_id = 0;
139  for (int c = 1; c < num_classes; ++c) {
140  if (pred[5 + c] > max_cls) {
141  max_cls = pred[5 + c];
142  class_id = c;
143  }
144  }
145 
146  float score = obj * max_cls;
147  if (score < conf_thresh) continue;
148 
149  dets.push_back({pred[0], pred[1], pred[2], pred[3], score, class_id});
150  }
151 
152  // NMS
153  std::vector<Detection> result;
154  std::sort(dets.begin(), dets.end(), [](const Detection& a, const Detection& b) {
155  return a.conf > b.conf;
156  });
157 
158  std::vector<bool> removed(dets.size(), false);
159  for (size_t i = 0; i < dets.size(); ++i) {
160  if (removed[i]) continue;
161  result.push_back(dets[i]);
162 
163  for (size_t j = i + 1; j < dets.size(); ++j) {
164  if (removed[j]) continue;
165 
166  // Calcular IoU
167  float xx1 = std::max(dets[i].x - dets[i].w/2, dets[j].x - dets[j].w/2);
168  float yy1 = std::max(dets[i].y - dets[i].h/2, dets[j].y - dets[j].h/2);
169  float xx2 = std::min(dets[i].x + dets[i].w/2, dets[j].x + dets[j].w/2);
170  float yy2 = std::min(dets[i].y + dets[i].h/2, dets[j].y + dets[j].h/2);
171 
172  float w = std::max(0.0f, xx2 - xx1);
173  float h = std::max(0.0f, yy2 - yy1);
174  float inter = w * h;
175  float area1 = dets[i].w * dets[i].h;
176  float area2 = dets[j].w * dets[j].h;
177  float ovr = inter / (area1 + area2 - inter);
178 
179  if (ovr > nms_thresh) removed[j] = true;
180  }
181  }
182  return result;
183 }
